1. Get the PostgreSQL service endpoint:
{{- if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "chart-library.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "PostgreSQL endpoint: ${NODE_IP}:${NODE_PORT}"
{{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "chart-library.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "PostgreSQL endpoint: ${SERVICE_IP}:{{ (index .Values.service.ports 0).port }}"
{{- else }}
  echo "PostgreSQL endpoint: {{ include "chart-library.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ (index .Values.service.ports 0).port }}"
{{- end }}
